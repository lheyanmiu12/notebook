dl标签

dl：自定义列表

Example

<body>
<!--
    第一张组合，常规组合
-->
<dl>
    <dt>我最爱玩的游戏</dt>
    <dd>吃鸡游戏</dd>
    <dt>我最爱上的网站</dt>
    <dd>CSDN程序猿之家</dd>
</dl>
<!--第二种组合，标题组合-->
<dl>
    <dt>新闻1</dt>
    <dt>新闻2</dt>
    <dt>新闻3</dt>
</dl>
<!--第三种组合-->
<dl>
        <dt>英雄内容</dt>
        <dd>你</dd>
        <dd>唔</dd>
        <dd>唔</dd>
</dl>
</body>
